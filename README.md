# My .dotfiles

## Get the config

    cd ~
    git clone git@github.com:rey/.dot.git
    ln -s ~/.dot/bash_profile ~/.bash_profile
    ln -s ~/.dot/bashrc ~/.bashrc
    ln -s ~/.dot/gitconfig ~/.gitconfig
    ln -s ~/.dot/gitconfig_global ~/.gitconfig_global
    ln -s ~/.dot/signature ~/.signature
    cd ~/.dot
